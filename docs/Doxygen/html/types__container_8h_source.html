297  virtual int FromJson(const FromJsonReq &req, IJST_OUT FromJsonResp &resp) IJSTI_OVERRIDE
298  {
299  IJSTI_RET_WHEN_TYPE_MISMATCH((req.stream.IsObject()), "object");
300 
301  assert(req.pFieldBuffer != IJST_NULL);
302  VarType& field = *static_cast<VarType *>(req.pFieldBuffer);
303  field.clear();
304  SerializerInterface& intf = IJSTI_FSERIALIZER_INS(T);
305 
306  for (rapidjson::Value::MemberIterator itMember = req.stream.MemberBegin();
307  itMember != req.stream.MemberEnd(); ++itMember)
308  {
309  const std::string fieldName(itMember->name.GetString(), itMember->name.GetStringLength());
310  // New a elem buffer in container first to avoid copy
311  typename VarType::value_type buf(fieldName, T());
312  std::pair<typename VarType::iterator, bool> insertRet = field.insert(IJSTI_MOVE(buf));
313  // Check duplicate
314  if (!insertRet.second) {
315  resp.errDoc.ElementMapKeyDuplicated(fieldName);
316  return ErrorCode::kDeserializeMapKeyDuplicated;
317  }
318 
319  // Element FromJson
320  T &elemBuffer = insertRet.first->second;
321  FromJsonReq elemReq(itMember->value, req.allocator,
322  req.deserFlag, req.canMoveSrc, &elemBuffer, 0);
323  FromJsonResp elemResp(resp.errDoc);
324  int ret = intf.FromJson(elemReq, elemResp);
325  if (ret != 0)
326  {
327  field.erase(fieldName);
328  resp.errDoc.ErrorInMap(fieldName);
329  return ret;
330  }
331  }
332  IJSTI_RET_WHEN_VALUE_IS_DEFAULT((field.empty()));
