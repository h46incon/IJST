335  IJSTI_RET_WHEN_TYPE_MISMATCH((req.stream.IsObject()), "object");

341

343  itMember != itEnd; ++itMember)
344  {
345  // New a elem buffer in container first to avoid copy
346  std::pair<typename VarType::iterator, bool> insertRet = field.insert(
347  std::pair<const std::basic_string<Ch>, T>(GetJsonStr(itMember->name), T()));
348  // Check duplicate

351  return ErrorCode::kDeserializeMapKeyDuplicated;
352  }
353 
354  // Element FromJson

356  FromJsonReq elemReq(itMember->value, req.allocator,
357  req.deserFlag, req.canMoveSrc, &elemBuffer, 0);
358  FromJsonResp elemResp(resp.errDoc);

360  if (ret != 0)
361  {
362  const std::basic_string<Ch> fieldName = GetJsonStr(itMember->name);
363  field.erase(fieldName);
364  resp.errDoc.ErrorInMap(fieldName);
