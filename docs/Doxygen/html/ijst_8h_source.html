1235  int DoSerializeFields(HandlerBase &writer, SerFlag::Flag serFlag, IJST_OUT rapidjson::SizeType& fieldCountOut) const
1236  {
1237  IJST_ASSERT(!m_isParentVal || m_pMetaClass->GetFieldsInfo().size() == 1);
1238  for (std::vector<MetaFieldInfo>::const_iterator itMetaField = m_pMetaClass->GetFieldsInfo().begin();
1239  itMetaField != m_pMetaClass->GetFieldsInfo().end(); ++itMetaField)
1240  {
1241  // Check field state
1242  const EFStatus fstatus = m_r->fieldStatus[itMetaField->index];
1243  switch (fstatus) {
1244  case FStatus::kMissing:
1245  if (detail::Util::IsBitSet(serFlag, SerFlag::kIgnoreMissing)) {
1246  continue;
1247  }
1248  // Fall through
1249  case FStatus::kValid:
1250  {
1251  const void *pFieldValue = GetFieldByOffset(itMetaField->offset);
1252  if (!m_isParentVal) {
1253  // write key
1254  IJSTI_RET_WHEN_WRITE_FAILD(
1255  writer.Key(itMetaField->jsonName.data(), (rapidjson::SizeType)itMetaField->jsonName.size()) );
1256  }
1257  // write value
1258  SerializeReq req(writer, pFieldValue, serFlag);
1259  IJSTI_RET_WHEN_NOT_ZERO(
1260  itMetaField->serializerInterface->Serialize(req));
1261  ++fieldCountOut;
1262  }
1263  break;
1264 
1265  case FStatus::kNull:
1266  {
1267  if (detail::Util::IsBitSet(serFlag, SerFlag::kIgnoreNull)) {
1268  continue;
1269  }
1270

1272  // write key
1273  IJSTI_RET_WHEN_WRITE_FAILD(

1275  }
1276  // write value
1277  IJSTI_RET_WHEN_WRITE_FAILD(writer.Null());
1278  ++fieldCountOut;
1279  }
1280  break;
1281 
1282  case FStatus::kNotAField:

1285  assert(false);

1287  }
1288  }

1304  int DoMoveFromJson(rapidjson::Value &stream, FromJsonParam& p)
1305  {
1306  if (m_isParentVal) {
1307  // Set field by stream itself
1308  assert(m_pMetaClass->GetFieldsInfo().size() == 1);
1309  return DoFieldFromJson(
1310  &m_pMetaClass->GetFieldsInfo()[0], stream, /*canMoveSrc=*/true, p);
1311  }
1312 
1313  // Set fields by members of stream
1314  if (!stream.IsObject()) {
1315  p.errDoc.ElementTypeMismatch("object", stream);
1316  return ErrorCode::kDeserializeValueTypeError;
1317  }
1318 
1319  // For each member
1320  rapidjson::Value::MemberIterator itNextRemain = stream.MemberBegin();
1321  for (rapidjson::Value::MemberIterator itMember = stream.MemberBegin();
1322  itMember != stream.MemberEnd(); ++itMember)
1323  {
1324 
1325  // Get related field info
1326  const std::string fieldName(itMember->name.GetString(), itMember->name.GetStringLength());
1327  const MetaFieldInfo *pMetaField = m_pMetaClass->FindFieldByJsonName(fieldName);
1328 
1329  if (pMetaField == IJST_NULL) {
1330  // Not a field in struct
1331  if (detail::Util::IsBitSet(p.deserFlag, DeserFlag::kErrorWhenUnknown)) {
1332  p.errDoc.UnknownMember(fieldName);
1333  return ErrorCode::kDeserializeSomeUnknownMember;
1334  }
1335  if (!detail::Util::IsBitSet(p.deserFlag, DeserFlag::kIgnoreUnknown)) {
1336  // Move unknown fields to the front of array first
1337  // TODO: This is relay on the implementation details of rapidjson's object storage (array), how to check?
1338  if (itNextRemain != itMember) {
1339  itNextRemain->name.SetNull().Swap(itMember->name);
1340  itNextRemain->value.SetNull().Swap(itMember->value);
1341  }
1342  ++itNextRemain;
1343  }
1344  continue;
1345  }
1346 
1347  // Move member out of object
1348  rapidjson::Value memberStream(rapidjson::kNullType);
1349  memberStream.Swap(itMember->value);
1350 
1351  IJSTI_RET_WHEN_NOT_ZERO(
1352  DoFieldFromJson(pMetaField, memberStream, /*canMoveSrc=*/true, p) );
1353  }
1354 
1355  // Clean deserialized
1356  if (stream.MemberCount() != 0) {
1357  stream.EraseMember(itNextRemain, stream.MemberEnd());
1358  }
1359  if (detail::Util::IsBitSet(p.deserFlag, DeserFlag::kIgnoreUnknown)) {
1360  m_r->unknown.SetObject();
1361  }
1362  else {
1363  m_r->unknown.SetNull().Swap(stream);
1364  }
1365 
1366  if (!detail::Util::IsBitSet(p.deserFlag, DeserFlag::kNotCheckFieldStatus)) {
1367  return CheckFieldState(p.errDoc);
1368  }
1369  else {
1370  return 0;
1371  }
