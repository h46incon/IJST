790  int DoSerialize(HandlerBase<Ch> &writer, SerFlag::Flag serFlag) const
791  {
792  rapidjson::SizeType fieldCount = 0;
793  if (m_r->isParentVal) {
794  return DoSerializeFields(writer, serFlag, fieldCount);
795  // Unknown will be ignored
796  }
797 
798  IJSTI_RET_WHEN_WRITE_FAILD(writer.StartObject());
799 
800  // Write fields
801  IJSTI_RET_WHEN_NOT_ZERO(DoSerializeFields(writer, serFlag, fieldCount));
802 
803  // Write buffer if need
804  if (!detail::Util::IsBitSet(serFlag, SerFlag::kIgnoreUnknown))
805  {
806  assert(m_r->unknown.IsObject());
807  for (typename TValue::ConstMemberIterator itMember = m_r->unknown.MemberBegin(), itEnd = m_r->unknown.MemberEnd();
808  itMember != itEnd; ++itMember)
809  {
810  // Write key
811  const TValue& key = itMember->name;
812  IJSTI_RET_WHEN_WRITE_FAILD(
813  writer.Key(key.GetString(), key.GetStringLength()) );
814  // Write value
815  IJSTI_RET_WHEN_WRITE_FAILD(
816  itMember->value.Accept(writer) );
817  }
818 
819  fieldCount += m_r->unknown.MemberCount();
820  }
821 
822  IJSTI_RET_WHEN_WRITE_FAILD(writer.EndObject(fieldCount));
823  return 0;
824  }

826  int DoSerializeFields(HandlerBase<Ch> &writer, SerFlag::Flag serFlag, IJST_OUT rapidjson::SizeType& fieldCountOut) const
827  {
828  IJST_ASSERT(!m_r->isParentVal || m_r->pMetaClass->GetFieldsInfo().size() == 1);
829  for (typename std::vector<TMetaFieldInfo>::const_iterator
830  itMetaField = m_r->pMetaClass->GetFieldsInfo().begin(), itEnd = m_r->pMetaClass->GetFieldsInfo().end();
831  itMetaField != itEnd; ++itMetaField)
832  {
833  // Check field state
834  const EFStatus fstatus = m_r->fieldStatus[itMetaField->index];
835  switch (fstatus) {
836  case FStatus::kMissing:
837  if (detail::Util::IsBitSet(serFlag, SerFlag::kIgnoreMissing)) {
838  continue;
839  }
840  // Fall through
841  case FStatus::kValid:
842  {
843  const void *pFieldValue = GetFieldByOffset(itMetaField->offset);
844  if (!m_r->isParentVal) {
845  // write key
846  IJSTI_RET_WHEN_WRITE_FAILD(
847  writer.Key(itMetaField->jsonName.data(), (rapidjson::SizeType)itMetaField->jsonName.size()) );
848  }
849  // write value
850  SerializeReq req(writer, pFieldValue, serFlag);
851  IJSTI_RET_WHEN_NOT_ZERO(
852  detail::GetSerializerInterface<Encoding>(*itMetaField)->Serialize(req));
853  ++fieldCountOut;
854  }
855  break;
856 
857  case FStatus::kNull:
858  {
859  if (detail::Util::IsBitSet(serFlag, SerFlag::kIgnoreNull)) {
860  continue;
861  }
862 
863  if (!m_r->isParentVal) {
864  // write key
865  IJSTI_RET_WHEN_WRITE_FAILD(
866  writer.Key(itMetaField->jsonName.data(), (rapidjson::SizeType)itMetaField->jsonName.size()) );
867  }
868  // write value
869  IJSTI_RET_WHEN_WRITE_FAILD(writer.Null());
870  ++fieldCountOut;
871  }
872  break;
873 
874  case FStatus::kNotAField:
875  default:
876  // Error occurs
877  assert(false);
878  return ErrorCode::kInnerError;
879  }
880  }
881  return 0;
882  }

913  for (typename TValue::MemberIterator itMember = stream.MemberBegin(), itEnd = stream.MemberEnd();

953  }

956  }
